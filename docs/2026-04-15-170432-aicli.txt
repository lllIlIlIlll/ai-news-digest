  一、项目结构

  ai-news-digest/
  ├── src/
  │   ├── index.ts          # 入口，CLI参数解析 + 主流程编排
  │   ├── fetcher.ts        # RSS抓取（并发拉取三个源）
  │   ├── parser.ts         # 文章解析（提取标题/链接/时间/来源）
  │   ├── filter.ts         # 24小时过滤 + 严格时间策略
  │   ├── formatter.ts      # Markdown日报生成
  │   └── output.ts         # 文件输出（latest + 日期归档）
  ├── output/               # 日报输出目录（gitkeep空目录）
  ├── package.json
  └── tsconfig.json

  二、核心流程

  启动 → 解析CLI参数 → 并发抓取3个RSS源
         ↓
     各源独立重试(最多3次)
         ↓
     解析文章 → 严格24小时过滤 → 按时间倒序
         ↓
     生成Markdown → 输出 latest.md + YYYY-MM-DD.md
         ↓
     汇总报告（成功N篇，失败N个源）

  三、各模块职责

  ┌──────────────┬────────────────────────────────────────────┐
  │     模块     │                    职责                    │
  ├──────────────┼────────────────────────────────────────────┤
  │ fetcher.ts   │ 拉取RSS XML，超时5秒/源，重试3次           │
  ├──────────────┼────────────────────────────────────────────┤
  │ parser.ts    │ 从XML解析标题、链接、发布时间、来源        │
  ├──────────────┼────────────────────────────────────────────┤
  │ filter.ts    │ 过滤24小时窗口，严格排除时间缺失的文章     │
  ├──────────────┼────────────────────────────────────────────┤
  │ formatter.ts │ 生成Markdown日报，包含统计摘要             │
  ├──────────────┼────────────────────────────────────────────┤
  │ output.ts    │ 同时写入 latest.md 和 output/YYYY-MM-DD.md │
  └──────────────┴────────────────────────────────────────────┘

  - 无其他运行时依赖

        console.warn(`\n  ⚠️ 摘要生成失败，跳过: ${article.title.slice(0, 
  30)}`);
      }
    }
    if (!quiet) process.stdout.write('\n');
  }

  async function generateSummary(title: string, description: string): 
  Promise<string> {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'x-api-key': API_KEY,
        'anthropic-version': '2023-06-01',
        'content-type': 'application/json',
      },
      body: JSON.stringify({
        model: MODEL,
        max_tokens: 150,
        messages: [{
          role: 'user',
          content: `为以下文章生成一句话中文摘要（不超过50字）：\n标题：${title}
  \n内容：${description.slice(0, 500)}`
        }]
      })
    });

    if (!response.ok) throw new Error(`API ${response.status}`);
    const data = await response.json();
    return data.content[0].text.trim();
  }
